Конструкторы, задача

Благодаря этому коду пёсик Шарик научится?

function Animal() {}

Animal.prototype.sleep = function () { return 'zzZZZ' };

function Cat() {}

Cat.prototype = Animal.prototype;
Cat.prototype.meow = function () { return 'Meow, meow, meow' };

function Dog() {}

Dog.prototype = Animal.prototype;
Dog.prototype.woof = function () { return 'Woof, woof, woof' };

var sharik = new Dog();

только спать
Ответ: no
